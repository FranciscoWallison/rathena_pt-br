valkyrie,53,54,3	script	Menestrel#Valkyrie	741,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Menestrel]";
			mes "Parabéns";
			mes "Honra aos guerreiros!";
			close;
		}
		mes "[Menestrel]";
		mes "Quer cantar comigo?";
		mes "Sha la la la la~";
		close;
	}
	else if (ADVJOB == Job_Clown && Class == Job_Archer_High && JobLevel > 39) {
		mes "[Menestrel]";
		mes "O mundo sombrio dos mortais precisa de uma música mais alegre.";
        mes "Você irá trazer a música para eles e virar a maré na batalha contra o mal?";
		next;
		mes "[Menestrel]";
		mes "Você fará isto por Rune-Midgard";
		mes "Como Menestrel?";
		next;
		if (select("Não.:Sim.") == 1) {
			mes "[Menestrel]";
			mes "Quando você estiver pronto, volte e fale comigo.";
			mes "Honra aos guerreiros!";
			close;
		}
		if (SkillPoint) {
			mes "[Menestrel]";
			mes "Você ainda tem pontos de habilidades disponíveis.";
			mes "Por favor, utilize todos, e torne a falar comigo.";
			close;
		}
		jobchange Job_Clown;
		set ADVJOB,0;
		mes "[Menestrel]";
		mes "Parabéns!";
		mes "Como um Menestrel, suas suas músicas trarão esperança para seus aliados e desespero para seus inimigos.";
		close;
	}
	mes "[Menestrel]";
	mes "Bem vindo";
	mes "Ao salão de Honra de Valhalla.";
	next;
	mes "[Menestrel]";
	mes "Por favor, sinta-se a vontade enquanto estiver por aqui.";
	mes "Honra aos Guerreiros!";
	close;
}
